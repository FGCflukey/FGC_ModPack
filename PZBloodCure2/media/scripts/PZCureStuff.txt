	{
		SyringeZblood=25,
		destroy PZGlassTestTube,
		destroy PZGlassFlask,
		
		Result:BloodCure,
		Time:1000.0,
		NeedToBeLearn:true,
		Category:PZBloodCure,
		OnCreate:Return_AllUsedItems,
	}
